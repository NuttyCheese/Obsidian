Задача 1
Конкуренция потоков

Надо исправить проблему с гонкой потоков.
```swift 
class Counter {
    private var count = 0

    func increment() {
        DispatchQueue.global().async {
            self.count += 1
        }
    }

    func getCount() -> Int {
        return count
    }
}

let counter = Counter()
for _ in 0..<1000 {
    counter.increment()
}
print("Count: \(counter.getCount())")
```

**Ожидаемое исправление:**
Использовать `DispatchQueue` для синхронизации доступа к `count`.

```swift
class Counter {
    private var count = 0
    private let queue = DispatchQueue(label: "counterQueue")

    func increment() {
        queue.async {
            self.count += 1
        }
    }

    func getCount() -> Int {
        return queue.sync {
            return self.count
        }
    }
}
```

 Задание 6: Параллельная обработка данных

Создай большой массив случайных чисел. Раздели массив на несколько частей и параллельно вычисли сумму каждой части. Объедини результаты и вычисли итоговую сумму.

 Подсказка:
Используй `DispatchQueue` и `DispatchGroup` для параллельной обработки данных.

```swift
let array = (0..<1000).map { _ in Int.random(in: 1...100) }
let chunkSize = 100
var results = [Int]()
let group = DispatchGroup()
let queue = DispatchQueue.global()
let lock = NSLock()

for i in stride(from: 0, to: array.count, by: chunkSize) {
    group.enter()
    queue.async {
        let chunk = Array(array[i..<min(i + chunkSize, array.count)])
        let sum = chunk.reduce(0, +)
        lock.lock()
        results.append(sum)
        lock.unlock()
        group.leave()
    }
}

group.notify(queue: .main) {
    let totalSum = results.reduce(0, +)
    print("Total sum: \(totalSum)")
}
```

 Задание 13: Создание потокобезопасного массива

Создай потокобезопасный массив, который может добавлять и удалять элементы параллельно. Используй `DispatchQueue` или `NSLock` для синхронизации доступа к массиву.

 Подсказка:
Используй `DispatchQueue.sync` или `NSLock` для обеспечения потокобезопасности.

```swift
class ThreadSafeArray<T> {
    private var array: [T] = []
    private let queue = DispatchQueue(label: "com.example.threadSafeArray", attributes: .concurrent)
    
    func append(_ element: T) {
        queue.async(flags: .barrier) {
            self.array.append(element)
        }
    }
    
    func remove(at index: Int) -> T? {
        var element: T?
        queue.sync {
            if index < self.array.count {
                element = self.array.remove(at: index)
            }
        }
        return element
    }
    
    var count: Int {
        var count = 0
        queue.sync {
            count = self.array.count
        }
        return count
    }
    
    subscript(index: Int) -> T? {
        var element: T?
        queue.sync {
            if index < self.array.count {
                element = self.array[index]
            }
        }
        return element
    }
}

let threadSafeArray = ThreadSafeArray<Int>()

DispatchQueue.concurrentPerform(iterations: 100) { index in
    threadSafeArray.append(index)
}

print("Count: \(threadSafeArray.count)")
```


Задание 14

Дан код. Нужно написать потокобезопасный аналог нативного NotificationCenter. То есть нужно реализовать функции add, remove, post, hasObserver 
```swift
protocol Observer {
 func observe(event: String, object: Any?, notificationCenter: ConcurrentNotificationCenter)
}

class ConcurrentNotificationCenter {
 static let shared = ConcurrentNotificationCenter()
 private init() {}

 func add(_ observer: Observer, forEvent event: String) { }

 func remove(_ observer: Observer, forEvent event: String) { }

 func post(event: String, object: Any?) { }

 func hasObserver(_ observer: Observer, forEvent event: String) -> Bool { }

}
```
Ответ
В коде нужно решить проблему читателей-писателей и правильно организовать хранение слушателей.

```swift
protocol Observer: AnyObject {
    func observe(event: String, object: Any?, notificationCenter: ConcurrentNotificationCenter)
}

class ConcurrentNotificationCenter {
    static let shared = ConcurrentNotificationCenter()

    private let observers: [String: NSHashTable<AnyObject>] = [:]
    private let syncQueue = DispatchQueue(label: "com.ConcurrentNotificationCenter", attributes: [.concurrent])

    private init() {}

    func add(_ observer: Observer, forEvent event: String) {
        syncQueue.async(flags: [.barrier]) {
            self.observers[event, default: NSHashTable.weakObjects()].add(observer)
        }
    }

    func remove(_ observer: Observer, forEvent event: String) {
        syncQueue.async(flags: [.barrier]) {
            self.observers[event]?.remove(observer)
        }
    }

    func post(event: String, object: Any?) {
        let observers = syncQueue.sync {
            return self.observers[event]?.allObjects as? [Observer]
        }
        observers?.forEach({ $0.observe(event: event, object: object, notificationCenter: self) })
    }

    func hasObserver(_ observer: Observer, forEvent event: String) -> Bool {
        syncQueue.sync {
            return observers[event]?.contains(observer) == true
        }
    }
}
```

Пояснения к решению:
В качестве хранения данных допускаются структуры:
 ▪ [String: [Observer]] - но здесь есть проблема с retain cycle (но нужно учесть, что для решения проблемы retain cycle удаление обсервера на deinit не подойдет, так как deinit не вызовется)
 ▪ [String: NSHashTable<AnyObject>] - можно хранить слабые ссылки
 ▪ [String: NSPointerArray<AnyObject>] - можно хранить слабые ссылки
 ▪ [String: [Wrapper]], где Wrapper это:

```swift
class Wrapper {
    weak var object: Observer?
}
```

Для синхронизации можно использовать:
 • простой лок NSLock, DispatchSemaphore, serial DispatchQueue
 • очередь с барьером

Использование DispatchWorkItem

- Делаем кнопку для старта задачи и кнопку для отмены задачи.
- Когда пользователь нажимает кнопку "Старт", начни выполнение задачи с задержкой в 5 секунд.
- Если нажимаем кнопку "Отмена" до завершения задачи, отменяем выполнение задачи.

```swift
import UIKit

class ViewController: UIViewController {
    @IBOutlet weak var statusLabel: UILabel!
    var workItem: DispatchWorkItem?

    override func viewDidLoad() {
        super.viewDidLoad()
    }

    @IBAction func startTask(_ sender: UIButton) {
        workItem = DispatchWorkItem {
            DispatchQueue.main.async {
                self.statusLabel.text = "Задача завершена!"
            }
        }

        if let workItem = workItem {
            DispatchQueue.global().asyncAfter(deadline: .now() + 5, execute: workItem)
        }
        
        statusLabel.text = "Задача начата..."
    }

    @IBAction func cancelTask(_ sender: UIButton) {
        workItem?.cancel()
        statusLabel.text = "Задача отменена"
    }
}
```

```swift
import Foundation

func processData() {
    let arraySize = 1000000
    let numbers = (0..<arraySize).map { _ in Int.random(in: 1...100) }
    let chunkSize = arraySize / 4
    
    var results: [Int] = Array(repeating: 0, count: 4)
    let dispatchGroup = DispatchGroup()
    
    for i in 0..<4 {
        dispatchGroup.enter()
        DispatchQueue.global().async {
            let start = i * chunkSize
            let end = start + chunkSize
            let sum = numbers[start..<end].reduce(0, +)
            results[i] = sum
            dispatchGroup.leave()
        }
    }
    
    dispatchGroup.notify(queue: .main) {
        let totalSum = results.reduce(0, +)
        print("Итоговая сумма: \(totalSum)")
    }
}

import Foundation

class Counter {
    private var count = 0
    private let lock = NSLock()
    
    func increment() {
        lock.lock()
        count += 1
        lock.unlock()
    }

        return count
    }
}

let counter = Counter()
let queue = DispatchQueue.global()
let group = DispatchGroup()

for _ in 0..<1000 {
    group.enter()
    queue.async {
        counter.increment()
        group.leave()
    }
}

group.notify(queue: .main) {
    print("Итоговый счетчик: \(counter.value)")
}
```
