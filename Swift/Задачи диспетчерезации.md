1. Создайте протокол Shape с методом area() -> Double.
2. Реализуйте два класса Circle и Rectangle, которые соответствуют протоколу Shape.

    func area() -> Double
}

class Circle: Shape {
    let radius: Double
    
    init(radius: Double) {
        self.radius = radius
    }

    }

 Задание 3: Исследование производительности
1. Напишите небольшую программу, которая создает массив из 1000 объектов типа Shape, включая объекты Circle и Rectangle.
2. Измерьте время выполнения метода area() для всех объектов в массиве с использованием статической и динамической диспетчеризации.
3. Сравните результаты и сделайте выводы о влиянии типа диспетчеризации на производительность.


```swift
import Foundation

// Протокол Shape
protocol Shape {
    func area() -> Double
}

// Класс Circle
class Circle: Shape {
    let radius: Double
    
    init(radius: Double) {
        self.radius = radius
    }
    
    func area() -> Double {
        return .pi * radius * radius
    }
}

// Класс Rectangle
class Rectangle: Shape {
    let width: Double
    let height: Double
    
    init(width: Double, height: Double) {
        self.width = width
        self.height = height
    }
    
    func area() -> Double {
        return width * height
    }
}

// Функция для измерения времени выполнения
func measureTime(for block: () -> Void) -> Double {
    let start = CFAbsoluteTimeGetCurrent()
    block()
    let end = CFAbsoluteTimeGetCurrent()
    return end - start
}

// Создаем массив из 1000 объектов Shape
var shapes: [Shape] = []
for _ in 0..<500 {
    shapes.append(Circle(radius: Double.random(in: 1...10)))
    shapes.append(Rectangle(width: Double.random(in: 1...10), height: Double.random(in: 1...10)))
}

// Измеряем время выполнения метода area() с использованием динамической диспетчеризации
let dynamicDispatchTime = measureTime {
    _ = shapes.map { $0.area() }
}

// Реализуем статическую диспетчеризацию с помощью методов area() вне протокола
extension Circle {
    func staticArea() -> Double {
        return .pi * radius * radius
    }
}

extension Rectangle {
    func staticArea() -> Double {
        return width * height
    }
}

// Измеряем время выполнения метода area() с использованием статической диспетчеризации
let staticDispatchTime = measureTime {
    _ = shapes.compactMap { shape -> Double? in
        if let circle = shape as? Circle {
            return circle.staticArea()
        } else if let rectangle = shape as? Rectangle {
            return rectangle.staticArea()
        } else {
            return nil
        }
    }
}

// Выводим результаты
print("Dynamic Dispatch Time: \(dynamicDispatchTime) seconds")
print("Static Dispatch Time: \(staticDispatchTime) seconds")
```

Задача 13
Обработка больших данных

- Делаем большой массив случайных чисел.
- Делим массив на несколько частей и параллельно вычисляем суммы каждой части.
- Объедяем результаты и вычисляем итоговую сумму.
- Можно использовать DispatchQueue и DispatchGroup для параллельной обработки.

import Foundation

// Функция для генерации большого массива случайных чисел
func generateRandomNumbers(count: Int, range: ClosedRange<Int>) -> [Int] {
    return (0..<count).map { _ in Int.random(in: range) }
}

// Параллельная функция для вычисления суммы частей массива
func parallelSum(numbers: [Int], chunkSize: Int) -> Int {
    let queue = DispatchQueue.global(qos: .userInitiated)
    let group = DispatchGroup()
    let totalChunks = (numbers.count + chunkSize - 1) / chunkSize
    var partialSums: [Int] = Array(repeating: 0, count: totalChunks)
    
    for chunkIndex in 0..<totalChunks {
        group.enter()
        queue.async {
            let start = chunkIndex * chunkSize
            let end = min(start + chunkSize, numbers.count)
            let chunk = Array(numbers[start..<end])
            let sum = chunk.reduce(0, +)
            partialSums[chunkIndex] = sum
            group.leave()
        }
    }
    
    group.wait()
    let totalSum = partialSums.reduce(0, +)
    return totalSum
}

// Генерация большого массива случайных чисел
let numbers = generateRandomNumbers(count: 1_000_000, range: 1...100)

// Установка размера куска для параллельной обработки
let chunkSize = 10_000

// Вычисление итоговой суммы с параллельной обработкой
let totalSum = parallelSum(numbers: numbers, chunkSize: chunkSize)

print("Total Sum: \(totalSum)")



Задача 14
Синхронизация доступа к общим ресурсам

- Делаем общий счетчик. Запускаем несколько потоков, каждый из которых увеличивает счетчик.
- Можно использовать DispatchSemaphore или NSLock для синхронизации доступа к счетчику.


import Foundation

class Counter {
    private var value = 0
    private let semaphore = DispatchSemaphore(value: 1)
    
    func increment() {
        semaphore.wait()
        value += 1
        semaphore.signal()
    }
    
    func getValue() -> Int {
        return value
    }
}

// Функция для запуска нескольких потоков
func performIncrementOperations(with counter: Counter, iterations: Int, threadCount: Int) {
    let group = DispatchGroup()
    let queue = DispatchQueue.global(qos: .userInitiated)
    
    for _ in 0..<threadCount {
        group.enter()
        queue.async {
            for _ in 0..<iterations {
                counter.increment()
            }
            group.leave()
        }
    }
    
    group.wait()
}

// Создаем счетчик и запускаем несколько потоков для инкрементации
let counter = Counter()
let iterations = 100_000
let threadCount = 10

performIncrementOperations(with: counter, iterations: iterations, threadCount: threadCount)

print("Final Counter Value with DispatchSemaphore: \(counter.getValue())")

    func area() -> Double
}

    let radius: Double

        return .pi * radius * radius
    }

    let side: Double
    
    func area() -> Double {
        return side * side
    }

    func area() -> Double {
        return 0
    }

    }

    }
