**Поиск в графах** — это фундаментальная группа алгоритмов, используемая для обхода вершин и рёбер графа, поиска путей, проверки связности, нахождения кратчайших путей и решения многих других задач.

### Основные типы графов, встречающихся в iOS-разработке

- Невзвешенные (все рёбра одинаковые)
- Взвешенные (есть стоимость/расстояние)
- Ориентированные / неориентированные
- Деревья (частный случай графа)
- Социальные графы (друзья, подписчики)
- Графы навигации (MapKit, маршруты)

### Основные алгоритмы поиска

1. **DFS** (поиск в глубину) — рекурсивный или стековый обход
2. **BFS** (поиск в ширину) — очередь, идеален для кратчайшего пути в невзвешенном графе
3. **Dijkstra** — кратчайший путь во взвешенном графе с неотрицательными весами
4. **A*** — эвристический поиск (с эвристикой, например, расстояние по прямой)
5. **Bellman-Ford** — кратчайший путь с отрицательными весами (без отрицательных циклов)

## 2. DFS — Поиск в глубину

```swift
func dfs(_ graph: AdjacencyList, start: Int) -> [Int] {
    var visited = Set<Int>()
    var result = [Int]()
    
    func dfsVisit(_ node: Int) {
        guard !visited.contains(node) else { return }
        visited.insert(node)
        result.append(node)
        
        for neighbor in graph[node] ?? [] {
            dfsVisit(neighbor)
        }
    }
    
    dfsVisit(start)
    return result
}

// Пример
let graph: AdjacencyList = [
    0: [1, 2],
    1: [0, 3],
    2: [0, 3],
    3: [1, 2, 4],
    4: [3]
]
print(dfs(graph, start: 0))  // [0, 1, 3, 2, 4] или другой порядок
```

### DFS с поиском пути

```swift
func dfsPath(_ graph: AdjacencyList, from start: Int, to target: Int) -> [Int]? {
    var visited = Set<Int>()
    var path = [Int]()
    
    func dfsVisit(_ node: Int) -> Bool {
        guard !visited.contains(node) else { return false }
        visited.insert(node)
        path.append(node)
        
        if node == target { return true }
        
        for neighbor in graph[node] ?? [] {
            if dfsVisit(neighbor) { return true }
        }
        
        path.removeLast()  // backtrack
        return false
    }
    
    return dfsVisit(start) ? path : nil
}
```

## 3. BFS — Поиск в ширину

```swift
func bfs(_ graph: AdjacencyList, start: Int) -> [Int] {
    var visited = Set<Int>()
    var queue = [start]
    var result = [Int]()
    
    visited.insert(start)
    
    while !queue.isEmpty {
        let node = queue.removeFirst()
        result.append(node)
        
        for neighbor in graph[node] ?? [] where !visited.contains(neighbor) {
            visited.insert(neighbor)
            queue.append(neighbor)
        }
    }
    
    return result
}
```

### BFS — кратчайший путь в невзвешенном графе

```swift
func shortestPathBFS(_ graph: AdjacencyList, from start: Int, to target: Int) -> [Int]? {
    var queue: [(node: Int, path: [Int])] = [(start, [start])]
    var visited = Set<Int>()
    
    visited.insert(start)
    
    while !queue.isEmpty {
        let (node, path) = queue.removeFirst()
        
        if node == target {
            return path
        }
        
        for neighbor in graph[node] ?? [] where !visited.contains(neighbor) {
            visited.insert(neighbor)
            queue.append((neighbor, path + [neighbor]))
        }
    }
    
    return nil
}

// Пример
print(shortestPathBFS(graph, from: 0, to: 4) ?? [])  // [0, 1, 3, 4] или [0, 2, 3, 4]
```

## 4. Dijkstra — кратчайший путь во взвешенном графе

```swift
func dijkstra(_ graph: WeightedAdjacencyList, from start: Int) -> [Int: Double] {
    var distances: [Int: Double] = [start: 0]
    var priorityQueue: [(node: Int, dist: Double)] = [(start, 0)]
    var visited = Set<Int>()
    
    while !priorityQueue.isEmpty {
        priorityQueue.sort { $0.dist < $1.dist }  // простой вариант (можно использовать Heap)
        let (node, currentDist) = priorityQueue.removeFirst()
        
        if visited.contains(node) { continue }
        visited.insert(node)
        
        for (neighbor, weight) in graph[node] ?? [] {
            let newDist = currentDist + weight
            if distances[neighbor] == nil || newDist < distances[neighbor]! {
                distances[neighbor] = newDist
                priorityQueue.append((neighbor, newDist))
            }
        }
    }
    
    return distances
}

// Пример
let weightedGraph: WeightedAdjacencyList = [
    0: [(1, 4), (2, 1)],
    1: [(3, 1)],
    2: [(1, 2), (3, 5)],
    3: []
]
print(dijkstra(weightedGraph, from: 0))  // [0: 0.0, 1: 3.0, 2: 1.0, 3: 4.0]
```

## 5. Реальные сценарии в [[iOS]]

### Пример 1: Поиск пути в дереве категорий

```swift
let categories: AdjacencyList = [
    0: [1, 2],      // Root
    1: [3, 4],      // Электроника
    2: [5],         // Одежда
    3: [], 4: [], 5: []
]

func findPathToCategory(from root: Int, to target: Int) -> [Int]? {
    shortestPathBFS(categories, from: root, to: target)
}
```

### Пример 2: Проверка связности графа (компоненты связности)

```swift
func findConnectedComponents(_ graph: AdjacencyList) -> [[Int]] {
    var visited = Set<Int>()
    var components: [[Int]] = []
    
    for node in graph.keys.sorted() {
        if !visited.contains(node) {
            let component = dfs(graph, start: node)
            components.append(component)
            visited.formUnion(component)
        }
    }
    
    return components
}
```

### Пример 3: Поиск цикла в ориентированном графе

```swift
func hasCycle(_ graph: AdjacencyList) -> Bool {
    var visited = Set<Int>()
    var recStack = Set<Int>()
    
    func dfsCycle(_ node: Int) -> Bool {
        visited.insert(node)
        recStack.insert(node)
        
        for neighbor in graph[node] ?? [] {
            if !visited.contains(neighbor) {
                if dfsCycle(neighbor) { return true }
            } else if recStack.contains(neighbor) {
                return true
            }
        }
        
        recStack.remove(node)
        return false
    }
    
    for node in graph.keys {
        if !visited.contains(node) && dfsCycle(node) {
            return true
        }
    }
    
    return false
}
```

## Итог — когда какой алгоритм выбрать

| Задача                              | Рекомендуемый алгоритм       | Сложность       |
|-------------------------------------|------------------------------|-----------------|
| Обход всех вершин                   | DFS или BFS                  | O(V + E)        |
| Кратчайший путь в невзвешенном графе | BFS                        | O(V + E)        |
| Кратчайший путь во взвешенном (≥ 0) | Dijkstra                   | O((V + E) log V)|
| Есть ли путь между вершинами        | BFS / DFS                    | O(V + E)        |
| Есть ли цикл                        | DFS с рекурсивным стеком     | O(V + E)        |
| Компоненты связности                | DFS / BFS несколько раз      | O(V + E)        |
| Навигация с эвристикой              | A*                           | O(E) в среднем  |

В iOS-разработке **BFS** и **DFS** покрывают большинство задач (игры, социальные графы, категории, навигация).  
Для полноценной навигации обычно используют **MapKit** или сторонние библиотеки (Graph, SwiftGraph, [[Alamofire]] и т.д.).
