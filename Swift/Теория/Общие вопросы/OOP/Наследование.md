## Наследование в Swift

### Определение
**Наследование** — это механизм объектно-ориентированного программирования, позволяющий создавать новый класс на основе существующего. Дочерний класс (подкласс) наследует все свойства и методы родительского класса (суперкласса) и может добавлять новые или переопределять существующие.

Наследование — один из трёх фундаментальных принципов ООП (наряду с инкапсуляцией и полиморфизмом). В Swift наследование доступно только для **классов** (структуры и перечисления не поддерживают наследование).

### Зачем это знать [[iOS]]-разработчику?
1.  **Переиспользование кода:** Общая логика выносится в базовый класс.
2.  **Полиморфизм:** Разные подклассы могут вести себя по-разному через переопределение методов.
3.  **Иерархии:** Создание иерархий типов (UIView → [[UIButton]], [[UILabel]], [[UIImageView]]).
4.  **Библиотеки и фреймворки:** [[UIKit]] построен на глубоком наследовании ([[UIResponder]] → [[UIView]] → [[UIViewController]]).
5.  **Базовые классы:** Создание абстрактных классов с общей логикой.

### Базовый синтаксис

```swift
class Animal {
    var name: String
    
    init(name: String) {
        self.name = name
    }
    
    func makeSound() {
        print("\(name) makes a sound")
    }
}

class Dog: Animal {
    // Наследует все свойства и методы Animal
    func fetch() {
        print("\(name) is fetching")
    }
}

let dog = Dog(name: "Buddy")
dog.makeSound()  // Buddy makes a sound (унаследован)
dog.fetch()      // Buddy is fetching (добавлен)
```

### Переопределение методов ([[override]])

```swift
class Animal {
    func makeSound() {
        print("Some generic sound")
    }
    
    func move() {
        print("Moving...")
    }
}

class Cat: Animal {
    // Переопределение метода
    override func makeSound() {
        print("Meow!")
    }
    
    // Вызов родительского метода
    override func move() {
        super.move()
        print("Cat is sneaking")
    }
}

let cat = Cat()
cat.makeSound()  // Meow!
cat.move()
// Moving...
// Cat is sneaking
```

### Переопределение свойств

```swift
class Vehicle {
    var wheels: Int {
        return 4
    }
    
    var description: String {
        return "Vehicle with \(wheels) wheels"
    }
}

class Bicycle: Vehicle {
    // Переопределение вычисляемого свойства
    override var wheels: Int {
        return 2
    }
    
    // Добавление наблюдателя
    override var description: String {
        return "Bicycle with \(wheels) wheels"
    }
}

let bike = Bicycle()
print(bike.wheels)      // 2
print(bike.description) // Bicycle with 2 wheels
```

### Инициализаторы и наследование

#### 1. **Designated и Convenience инициализаторы**

```swift
class Person {
    var name: String
    var age: Int
    
    // Designated инициализатор
    init(name: String, age: Int) {
        self.name = name
        self.age = age
    }
    
    // Convenience инициализатор
    convenience init(name: String) {
        self.init(name: name, age: 0)
    }
}

class Student: Person {
    var studentId: String
    
    // Переопределение designated инициализатора
    override init(name: String, age: Int) {
        self.studentId = "unknown"
        super.init(name: name, age: age)
    }
    
    // Свой инициализатор
    init(name: String, age: Int, studentId: String) {
        self.studentId = studentId
        super.init(name: name, age: age)
    }
}

let student = Student(name: "Alice", age: 20, studentId: "12345")
```

#### 2. **Требуемые инициализаторы (required)**

```swift
class Shape {
    var color: String
    
    required init(color: String) {
        self.color = color
    }
}

class Circle: Shape {
    var radius: Double
    
    init(radius: Double, color: String) {
        self.radius = radius
        super.init(color: color)
    }
    
    // required инициализатор должен быть реализован
    required init(color: String) {
        self.radius = 0
        super.init(color: color)
    }
}
```

### Запрет на переопределение ([[final]])

```swift
class Parent {
    // final метод — нельзя переопределить
    final func cannotOverride() {
        print("This method is final")
    }
    
    // final свойство — нельзя переопределить
    final var value: Int = 42
    
    // final класс — от него нельзя наследовать
}

class Child: Parent {
    // ❌ Ошибка: cannot override final method
    // override func cannotOverride() { }
}

// ❌ Ошибка: нельзя наследовать от final класса
// class GrandChild: FinalClass { }
```

### Многоуровневое наследование

```swift
class Animal {
    func breathe() {
        print("Breathing")
    }
}

class Mammal: Animal {
    func giveBirth() {
        print("Giving birth")
    }
}

class Dog: Mammal {
    override func breathe() {
        print("Dog breathing")
    }
    
    func bark() {
        print("Woof!")
    }
}

let dog = Dog()
dog.breathe()    // Dog breathing (переопределён)
dog.giveBirth()  // Giving birth (унаследован от Mammal)
dog.bark()       // Woof! (свой)
```

### Наследование и [[полиморфизм]]

```swift
class Shape {
    func area() -> Double {
        return 0
    }
}

class Rectangle: Shape {
    var width: Double
    var height: Double
    
    init(width: Double, height: Double) {
        self.width = width
        self.height = height
    }
    
    override func area() -> Double {
        return width * height
    }
}

class Circle: Shape {
    var radius: Double
    
    init(radius: Double) {
        self.radius = radius
    }
    
    override func area() -> Double {
        return Double.pi * radius * radius
    }
}

let shapes: [Shape] = [Rectangle(width: 2, height: 3), Circle(radius: 2)]
for shape in shapes {
    print(shape.area())  // 6.0, 12.56
}
```

### Абстрактные классы (эмуляция)

[[Swift]] не имеет ключевого слова `abstract`, но можно эмулировать через `fatalError` или `required` инициализаторы.

```swift
class AbstractDataParser {
    func parse(data: Data) -> [String: Any] {
        fatalError("Subclasses must override parse(data:)")
    }
    
    func validate(data: Data) -> Bool {
        // Базовая реализация может быть переопределена
        return !data.isEmpty
    }
}

class JSONParser: AbstractDataParser {
    override func parse(data: Data) -> [String: Any] {
        // Реальная реализация парсинга JSON
        return [:]
    }
}

// let parser = AbstractDataParser()  // Опасно — вызовет fatalError
```

### Наследование vs Композиция

| Характеристика | Наследование | Композиция |
|----------------|--------------|------------|
| **Отношение** | "is-a" (является) | "has-a" (имеет) |
| **Связанность** | Высокая | Низкая |
| **Гибкость** | Низкая | Высокая |
| **Переиспользование** | Вертикальное | Горизонтальное |
| **Изменение поведения** | Через override | Через замену компонента |
| **Время жизни** | Вместе с объектом | Может быть независимым |

#### Пример композиции вместо наследования

```swift
// ❌ Наследование (жёсткая связь)
class MovingCar: Car {
    override func move() {
        // сложная логика движения
    }
}

// ✅ Композиция (гибкая связь)
protocol Movable {
    func move()
}

class Car {
    var mover: Movable
    
    init(mover: Movable) {
        self.mover = mover
    }
    
    func move() {
        mover.move()
    }
}

class FastMover: Movable {
    func move() {
        print("Moving fast")
    }
}

class SlowMover: Movable {
    func move() {
        print("Moving slow")
    }
}

let fastCar = Car(mover: FastMover())
let slowCar = Car(mover: SlowMover())
```

### Наследование в [[UIKit]]

```swift
// Иерархия UIView
// UIResponder → UIView → UIControl → UIButton
//                          → UILabel
//                          → UIImageView
//                          → UIScrollView → UITableView

class CustomButton: UIButton {
    override init(frame: CGRect) {
        super.init(frame: frame)
        setup()
    }
    
    required init?(coder: NSCoder) {
        super.init(coder: coder)
        setup()
    }
    
    private func setup() {
        backgroundColor = .blue
        setTitleColor(.white, for: .normal)
        layer.cornerRadius = 8
    }
    
    override func touchesBegan(_ touches: Set<UITouch>, with event: UIEvent?) {
        super.touchesBegan(touches, with: event)
        // Кастомная анимация
        UIView.animate(withDuration: 0.1) {
            self.alpha = 0.7
        }
    }
    
    override func touchesEnded(_ touches: Set<UITouch>, with event: UIEvent?) {
        super.touchesEnded(touches, with: event)
        UIView.animate(withDuration: 0.1) {
            self.alpha = 1.0
        }
    }
}
```

### Антипаттерны наследования

#### 1. **Глубокие иерархии**

```swift
// ❌ Плохо — слишком глубокая иерархия
class A { }
class B: A { }
class C: B { }
class D: C { }
class E: D { }
class F: E { }

// ✅ Хорошо — плоская иерархия с композицией
class CoreLogic { }
class FeatureA { private let core = CoreLogic() }
class FeatureB { private let core = CoreLogic() }
```

#### 2. **Нарушение принципа подстановки Лисков ([[Liskov Substitution Principle|LSP]])**

```swift
// ❌ Плохо — подкласс изменяет поведение родителя
class Bird {
    func fly() {
        print("Flying")
    }
}

class Penguin: Bird {
    override func fly() {
        fatalError("Penguins can't fly!")  // Нарушение LSP
    }
}

// ✅ Хорошо — пересмотр иерархии
protocol Flyable {
    func fly()
}

class Bird { }
class Eagle: Bird, Flyable {
    func fly() { print("Eagle flying") }
}
class Penguin: Bird { }
```

#### 3. **Наследование ради переиспользования кода**

```swift
// ❌ Плохо — наследование только ради метода
class Utility {
    func formatDate(date: Date) -> String { ... }
}

class MyViewController: Utility {  // UIViewController не должен наследовать Utility
    // ...
}

// ✅ Хорошо — композиция
class MyViewController: UIViewController {
    private let formatter = DateFormatter()
}
```

### Преимущества наследования

| Преимущество | Описание |
|--------------|----------|
| **Переиспользование кода** | Общая логика выносится в базовый класс |
| **Полиморфизм** | Один интерфейс — много реализаций |
| **Иерархии типов** | Естественное моделирование отношений "is-a" |
| **Расширяемость** | Легко добавлять новые подклассы |
| **Базовые классы** | Создание абстракций с общей логикой |

### Недостатки

| Недостаток | Описание |
|------------|----------|
| **Высокая связанность** | Изменение в родителе может сломать потомков |
| **Глубокие иерархии** | Сложность понимания и поддержки |
| **Хрупкий базовый класс** | Проблемы при изменении базового класса |
| **Наследование vs композиция** | Часто композиция предпочтительнее |
| **Только классы** | В Swift наследование только для классов |

---

### Короткое правило

> **Наследование** — отношение "is-a".  
> Используйте для полиморфизма и общих интерфейсов.  
> Предпочитайте **композицию** переиспользованию кода через наследование.

---

### Итог

**Наследование** в Swift:

1.  **Доступно только для классов**
2.  **Позволяет** переопределять методы и свойства (`override`)
3.  **Поддерживает** полиморфизм
4.  **Имеет** модификаторы `final` (запрет переопределения) и `required` (обязательная реализация)
5.  **Требует** вызова `super.init()` в designated инициализаторах
6.  **Часто заменяется композицией** для гибкости

Понимание наследования необходимо для работы с UIKit и создания иерархий типов, но важно не злоупотреблять им и помнить о композиции как альтернативе.
