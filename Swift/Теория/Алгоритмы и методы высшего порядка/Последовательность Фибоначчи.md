```swift
/// Возвращает первые `count` чисел Фибоначчи типа `Int`.
/// - Parameter count: сколько чисел вернуть (если <= 0 — вернётся пустой массив).
/// - Returns: массив длины `count`.
/// - Note: на 64-битной платформе `Int` ~ `Int64`. Безопасно генерировать до 92-го числа включительно:
///         F(92) = 7_540_113_804_746_346_429 <= Int64.max, а F(93) уже переполнит `Int`.
func fibonacciSequenceInt(_ count: Int) -> [Int] {
    guard count > 0 else { return [] }
    
    return sequence(first: (0, 1)) { ($0.1, $0.0 + $0.1) }
	    .prefix(count)
	    .map { $0.0 }
}
```

## 2. Итеративный (обычный) — [[Int]]

```swift
/// Итеративная реализация Фибоначчи.
/// - Parameter count: количество значений (если <= 0 — вернётся пустой массив).
/// - Returns: массив первых `count` чисел Фибоначчи.
func fibonacciIterativeInt(_ count: Int) -> [Int] {
    guard count > 0 else { return [] }
    precondition(count <= 92, "count too large: fibonacci numbers will overflow Int. Use smaller count or BigInt.")
    
    if count == 1 { return [0] }
    
    var result: [Int] = []
    result.reserveCapacity(count)
    
    var a = 0
    var b = 1
    
    result.append(a)           // 0
    for _ in 1..<count {       // добавляем b count-1 раз
        result.append(b)       // 1, 1, 2...
        let next = a &+ b      // можно использовать &+ чтобы позволить явное поведение при переполнении,
                               // но мы сделали precondition, поэтому используем обычное +
        a = b
        b = next
    }
    return result
}
```

> Примечание: я использовал `precondition` — это удобный способ явно упомянуть ограничение. Альтернатива — возвращать [[nil]] или [[Result]] при переполнении — могу показать, если нужно.

## 3. Реактивный (sequence-based) — [[Double]]

```swift
/// То же самое для Double.
/// - Важно: Double может хранить очень большие значения, но потеря точности (integer rounding)
///   начнёт происходить, когда числа станут больше ~2^53 (~9e15). Для Фибоначчи это примерно
///   начиная с F(79) — F(78) ещё обычно точно представимо в Double, F(79) уже может терять точность.
func fibonacciSequenceDouble(_ count: Int) -> [Double] {
    guard count > 0 else { return [] }
    
    return sequence(first: (0.0, 1.0), next: { (prev: (Double, Double)) -> (Double, Double) in
        (prev.1, prev.0 + prev.1)
    })
    .prefix(count)
    .map { $0.0 }
}
```

## 4. Итеративный (обычный) — `Double`

```swift
/// Итеративная версия для Double.
func fibonacciIterativeDouble(_ count: Int) -> [Double] {
    guard count > 0 else { return [] }
    if count == 1 { return [0.0] }
    
    var result: [Double] = []
    result.reserveCapacity(count)
    
    var a: Double = 0.0
    var b: Double = 1.0
    
    result.append(a)
    for _ in 1..<count {
        result.append(b)
        let next = a + b
        a = b
        b = next
    }
    return result
}
```

## 5. Пошаговый разбор (пример для `count = 7`)

### 5.1. Что делает [[sequence func|Sequence]]-вариант ([[tuple]] approach)

Мы используем последовательность состояний `(prev, curr)`:

- `first` = `(0, 1)` — это первое состояние. `map { $0.0 }` вернёт `0`.
    
- `next` для `(0,1)` возвращает `(1, 0+1) = (1,1)` — теперь `map` даст `1`.
    
- `next` для `(1,1)` → `(1, 1+1) = (1,2)` — даёт `1`.
    
- `next` для `(1,2)` → `(2, 1+2) = (2,3)` — даёт `2`.
    
- и т.д.
    

Таблица (tuple, output):

|step|tuple `(prev,curr)`|output (`$0.0`)|
|---|---|---|
|0|(0, 1)|0|
|1|(1, 1)|1|
|2|(1, 2)|1|
|3|(2, 3)|2|
|4|(3, 5)|3|
|5|(5, 8)|5|
|6|(8,13)|8|

Результат: `[0,1,1,2,3,5,8]`.

### 5.2. Что делает итеративный вариант

Инициализируем `a = 0`, `b = 1`, `result = [0]`.

Цикл `for _ in 1..<count` (count = 7 => итерации для _ = 1..6):

|итерация|до append (a,b)|append(b) -> result|next = a + b|обновление a=b, b=next|
|---|---|---|---|---|
|1|(0,1)|[0,1]|1|a=1, b=1|
|2|(1,1)|[0,1,1]|2|a=1, b=2|
|3|(1,2)|[0,1,1,2]|3|a=2, b=3|
|4|(2,3)|[0,1,1,2,3]|5|a=3, b=5|
|5|(3,5)|[0,1,1,2,3,5]|8|a=5, b=8|
|6|(5,8)|[0,1,1,2,3,5,8]|13|a=8, b=13|

Итог: `[0,1,1,2,3,5,8]` — точно тот же результат.

## 6. Сложность и память

- **Время:** O(n) — обе реализации делают `count` шагов (или `count-1` в цикле).
    
- **Память:** O(n) для выходного массива; дополнительная память O(1) (только несколько переменных/кортеж).

## 7. Переполнение (`Int`) и точность (`Double`)

### `Int` (64-бит)

- На 64-битных системах `Int` ≡ `Int64`.
    
- `F(92) = 7_540_113_804_746_346_429` — **всё ещё ≤ Int64.max**.
    
- `F(93) = 12_200_160_415_121_876_738` — **превышает** Int64.max → **переполнение**.
    
- Поэтому я поставил `precondition(count <= 92)` в `Int`-вариантах. Альтернативы:
    
    - возвращать `nil`/`Result` при превышении;
        
    - использовать `UInt64` (он вмещает F(93), но F(94) уже превысит `UInt64.max`);
        
    - использовать библиотеку произвольной точности (`BigInt` / `BigInteger`).
        

### `Double`

- `Double` может представлять очень большие числа (до ~1e308), но **целая точность** сохраняется только до ~`2^53 ≈ 9.007199e15`.
    
- Для последовательности Фибоначчи это означает: примерно до `F(78)` / `F(79)` ещё адекватно — начиная где-то с `F(79)` вы начнёте терять точную целую репрезентацию (т.е. числа будут «округлёнными»).
    
- Если нужен **точный целочисленный результат для больших n**, используйте `BigInt` (библиотеки) или собственную реализацию на [[Array]]/[[String]].

## 8. Доп. полезные фрагменты

### Безопасный вариант, возвращающий [[nil]] при переполнении ([[Int]])

```swift
/// Возвращает nil, если count слишком велик и будет переполнение Int.
func fibonacciSafeInt(_ count: Int) -> [Int]? {
    guard count > 0 else { return [] }
    guard count <= 92 else { return nil } // безопасная граница для Int64
    return fibonacciIterativeInt(count)
}
```

### Детекция переполнения во время вычисления (без precondition)

```swift
func fibonacciDetectOverflow(_ count: Int) -> [Int]? {
    guard count > 0 else { return [] }
    if count == 1 { return [0] }
    var result: [Int] = []
    result.reserveCapacity(count)
    var a = 0
    var b = 1
    result.append(a)
    for _ in 1..<count {
        result.append(b)
        let (sum, overflow) = a.addingReportingOverflow(b)
        if overflow { return nil } // обнаружили переполнение
        a = b
        b = sum
    }
    return result
}
```

## 10. Итог / Рекомендации

- Для компактного, «функционального» стиля — `sequence(...).prefix(count).map { $0.0 }` — красиво и читабельно.
    
- Для контроля над производительностью и предсказуемым поведением используйте итеративный вариант.
    
- Если нужен **большой индекс** (точные целые числа > F(92)), переходите на `BigInt` / библиотеку произвольной точности.
    
- Если используете `Double`, помните про потерю точности уже начиная примерно с ~F(79).
