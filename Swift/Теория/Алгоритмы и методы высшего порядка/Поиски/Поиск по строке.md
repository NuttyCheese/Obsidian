**Поиск по строке** — одна из самых частых операций в [[iOS]]-разработке: от фильтрации таблиц и поиска в текстовых полях до парсинга логов, [[API]]-ответов и пользовательского ввода.

[[Swift]] предоставляет **очень удобные и эффективные встроенные методы** для поиска подстрок, символов и паттернов.

## 1. Основные встроенные методы [[String]]

```swift
let text = "Hello, Swift developers! Welcome to iOS 18 in 2026."

// 1. Проверка наличия подстроки
text.contains("Swift")              // true
text.contains("swift")              // false (чувствителен к регистру)

// 2. Поиск первого вхождения (возвращает Range<String.Index>?)
if let range = text.range(of: "Swift") {
    let startIndex = text.distance(from: text.startIndex, to: range.lowerBound)
    print("Swift начинается с позиции \(startIndex)")     // 7
}

// 3. Поиск с игнорированием регистра
text.range(of: "swift", options: .caseInsensitive) != nil   // true

// 4. Поиск последнего вхождения
if let lastRange = text.range(of: "iOS", options: .backwards) {
    print("Последний 'iOS' начинается с \(text.distance(from: text.startIndex, to: lastRange.lowerBound))")
}

// 5. Все вхождения подстроки (расширение)
extension String {
    func indices(of substring: String, options: String.CompareOptions = []) -> [Int] {
        var indices: [Int] = []
        var searchStart = startIndex
        
        while let range = self[searchStart...].range(of: substring, options: options) {
            let index = distance(from: startIndex, to: range.lowerBound)
            indices.append(index)
            searchStart = range.upperBound
        }
        
        return indices
    }
}

print(text.indices(of: "e"))           // [1, 18, 20, 26, 35, 41]
print(text.indices(of: "Swift"))       // [7]
```

## 2. Поиск по символам ([[Character]])

```swift
let greeting = "Привет, мир! Hello, world! 🌍"

// Все позиции символа "!"
let exclamations = greeting.enumerated()
    .filter { $0.element == "!" }
    .map { $0.offset }

print(exclamations)  // [12, 25]

// Все эмодзи
let emojis = greeting.enumerated()
    .filter { $0.element.isEmoji }
    .map { $0.offset }

print(emojis)  // [27] (позиция 🌍)
```

## 3. Фильтрация слов и массивов строк

```swift
let words = [
    "SwiftUI", "UIKit", "Combine", "SwiftData",
    "RealityKit", "MapKit", "CoreData", "Swift"
]

// 1. Начинается с "Swift"
let swiftWords = words.filter { $0.hasPrefix("Swift") }
// → ["SwiftUI", "SwiftData", "Swift"]

// 2. Содержит "Kit"
let kitWords = words.filter { $0.contains("Kit") }
// → ["SwiftUI", "UIKit", "RealityKit", "MapKit"]

// 3. Заканчивается на "Kit" (игнорируя регистр)
let endingKit = words.filter { $0.lowercased().hasSuffix("kit") }
// → ["SwiftUI", "UIKit", "RealityKit", "MapKit"]

// 4. Поиск с регуляркой (NSRegularExpression или Regex в Swift 5.7+)
if #available(iOS 16.0, *) {
    let regex = try! Regex("[A-Z][a-z]+Kit")
    let capitalizedKit = words.filter { $0.firstMatch(of: regex) != nil }
    // → ["SwiftUI", "UIKit", "RealityKit", "MapKit"]
}
```

## 4. Поиск в [[UITextField]] / [[UITextView]] (реальный [[iOS]]-кейс)

```swift
// Фильтрация массива по поисковому запросу из search bar
func filterItems(_ items: [String], searchText: String) -> [String] {
    guard !searchText.isEmpty else { return items }
    
    let lowerSearch = searchText.lowercased()
    
    return items.filter { item in
        item.lowercased().contains(lowerSearch) ||
        item.lowercased().hasPrefix(lowerSearch)
    }
}

// Пример использования в UISearchController
func updateSearchResults(for searchController: UISearchController) {
    guard let text = searchController.searchBar.text else { return }
    filteredItems = filterItems(allItems, searchText: text)
    tableView.reloadData()
}
```

## 5. Поиск в [[Dictionary]] и сложных структурах

```swift
let developers = [
    "Anna": "SwiftUI & Combine expert",
    "Ben":  "iOS & macOS developer",
    "Clara": "RealityKit & AR enthusiast",
    "David": "Backend & Swift lover"
]

// 1. Найти всех, кто работает со Swift
let swiftDevs = developers.filter { $0.value.lowercased().contains("swift") }.keys
print(swiftDevs)  // ["Anna", "David", "Ben"]

// 2. Первый разработчик с AR
if let arDev = developers.first(where: { $0.value.contains("RealityKit") }) {
    print("AR-разработчик: \(arDev.key)")
}
```

## 6. Продвинутые примеры (парсинг, выделение)

```swift
// 1. Найти все email в тексте
let textWithEmails = """
Contact us: support@example.com or hello@company.io
Also: test.user+tag@gmail.com
"""

let emailRegex = try! Regex(#"[A-Z0-9a-z._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}"#)
let emails = textWithEmails.matches(of: emailRegex).map { String($0.0) }
print(emails)  // ["support@example.com", "hello@company.io", "test.user+tag@gmail.com"]

// 2. Выделить все числа
let log = "Error 404 at 14:35:12, retry #3, code 500"
let numbers = log.components(separatedBy: CharacterSet.decimalDigits.inverted)
    .filter { !$0.isEmpty }
    .compactMap { Int($0) }
print(numbers)  // [404, 14, 35, 12, 3, 500]
```

## 7. Сравнение производительности и выбор метода

| Задача                              | Лучший метод в Swift                     | Сложность     |
|-------------------------------------|------------------------------------------|---------------|
| Есть ли подстрока                   | `contains(_:)`                           | O(n)          |
| Индекс первого вхождения            | `range(of:)` / `firstIndex(of:)`         | O(n)          |
| Все индексы подстроки               | `indices(of:)` (расширение)              | O(n)          |
| Поиск без учёта регистра            | `range(of:options: .caseInsensitive)`    | O(n)          |
| Фильтрация массива строк            | `filter { $0.contains(...) }`            | O(n × m)      |
| Поиск по регулярке                  | `Regex` (iOS 16+) или `NSRegularExpression` | O(n) в среднем |
| Поиск в маленьком тексте            | любой встроенный метод                   | —             |

## Итог — современные рекомендации (2026)

| Ситуация                                  | Что использовать                                  |
| ----------------------------------------- | ------------------------------------------------- |
| Простая проверка наличия                  | [[contains]]`(_:)`                                |
| Нужен индекс первого вхождения            | `range(of:)` или [[firstIndex]]`(of:)`            |
| Нужны все вхождения                       | расширение `indices(of:)`                         |
| Поиск без учёта регистра                  | `options: .caseInsensitive`                       |
| Фильтрация списка по тексту из search bar | `filter { $0.lowercased().contains(...) }`        |
| Сложные паттерны (email, телефон, дата)   | `Regex` (Swift 5.7+)                              |
| Очень большой текст + частые запросы      | можно рассмотреть KMP / Boyer-Moore (редко нужно) |

В 95% случаев в [[iOS]]-разработке **встроенных методов [[String]]** + функции высшего порядка ([[filter]], [[first]], `contains`) полностью достаточно и очень эффективны.
