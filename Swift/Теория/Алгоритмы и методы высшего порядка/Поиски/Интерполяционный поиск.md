# Интерполяционный поиск (Interpolation Search) в [[Swift]]

**Интерполяционный поиск** — это улучшенная версия бинарного поиска, которая использует **линейную интерполяцию** для предсказания позиции элемента в **отсортированном числовом массиве**.

### Формула позиции

```text
pos = low + ((target - array[low]) × (high - low)) ÷ (array[high] - array[low])
```

Если массив равномерно распределён (арифметическая прогрессия, равномерные идентификаторы и т.д.), поиск может достигать **O(log log n)** в среднем случае.

### Когда он действительно быстрее бинарного поиска

- Данные близки к **равномерному распределению** (например: 10, 20, 30, 40, … или ID 1000, 1001, 1002, …)
- Массив **очень большой** (сотни тысяч — миллионы элементов)
- Значения — **целые числа** или числа с равномерным шагом

В худшем случае (например, экспоненциальное распределение) деградирует до **O(n)** — хуже бинарного.

## 1. Классическая итеративная реализация

```swift
/// Возвращает индекс первого найденного элемента или nil
func interpolationSearch(_ array: [Int], target: Int) -> Int? {
    guard !array.isEmpty else { return nil }
    
    var low = 0
    var high = array.count - 1
    
    while low <= high && target >= array[low] && target <= array[high] {
        // Защита от деления на ноль
        if array[low] == array[high] {
            return array[low] == target ? low : nil
        }
        
        // Интерполяционная формула
        let pos = low + (target - array[low]) * (high - low) / (array[high] - array[low])
        
        // Защита от выхода за границы (редко, но бывает при неточном распределении)
        let safePos = max(low, min(pos, high))
        
        if array[safePos] == target {
            return safePos
        } else if array[safePos] < target {
            low = safePos + 1
        } else {
            high = safePos - 1
        }
    }
    
    return nil
}
```

## 2. Версия с KeyPath (для массивов структур)

```swift
extension Array where Element: RandomAccessCollection {
    func interpolationSearch<Key: BinaryInteger>(
        by keyPath: KeyPath<Element, Key>,
        target: Key
    ) -> Int? where Element.Index == Int {
        guard !isEmpty else { return nil }
        
        var low = 0
        var high = count - 1
        
        while low <= high && target >= self[low][keyPath: keyPath] && target <= self[high][keyPath: keyPath] {
            if self[low][keyPath: keyPath] == self[high][keyPath: keyPath] {
                return self[low][keyPath: keyPath] == target ? low : nil
            }
            
            let pos = low + Int(target - self[low][keyPath: keyPath]) * (high - low) / Int(self[high][keyPath: keyPath] - self[low][keyPath: keyPath])
            let safePos = max(low, min(pos, high))
            
            let midValue = self[safePos][keyPath: keyPath]
            
            if midValue == target {
                return safePos
            } else if midValue < target {
                low = safePos + 1
            } else {
                high = safePos - 1
            }
        }
        
        return nil
    }
}
```

## 3. Реальные примеры использования

### Пример 1 — равномерный массив цен

```swift
let prices = [100, 150, 200, 250, 300, 350, 400, 450, 500, 550, 600]
if let index = interpolationSearch(prices, target: 350) {
    print("Цена 350 найдена на позиции \(index)") // → 5
}
```

### Пример 2 — поиск по ID пользователей

```swift
struct User {
    let id: Int
    let name: String
}

let users = (1000...1100).map { User(id: $0, name: "User\($0)") }

if let index = users.interpolationSearch(by: \.id, target: 1075) {
    print("Найден пользователь: \(users[index].name)") // User1075
}
```

### Пример 3 — поиск ближайшего значения (модификация)

```swift
func findClosest(_ array: [Int], target: Int) -> Int? {
    var low = 0
    var high = array.count - 1
    var closest = array[0]
    var minDiff = abs(target - array[0])
    
    while low <= high {
        if array[low] == array[high] { break }
        
        let pos = low + (target - array[low]) * (high - low) / (array[high] - array[low])
        let safePos = max(low, min(pos, high))
        
        let diff = abs(target - array[safePos])
        if diff < minDiff {
            minDiff = diff
            closest = array[safePos]
        }
        
        if array[safePos] < target {
            low = safePos + 1
        } else {
            high = safePos - 1
        }
    }
    
    return closest
}

let temperatures = [0, 5, 10, 15, 20, 25, 30, 35, 40]
print(findClosest(temperatures, target: 23) ?? -1) // → 25 или 20
```

### Пример 4 — поиск в массиве timestamp’ов (логи)

```swift
let timestamps = stride(from: 1700000000, to: 1700100000, by: 100).map { $0 }

if let index = interpolationSearch(timestamps, target: 1700050000) {
    print("Найден ближайший timestamp: \(timestamps[index])")
}
```

### Пример 5 — обработка повторяющихся значений

```swift
let scores = [10, 20, 30, 30, 30, 40, 50, 50, 60]

if let first = interpolationSearch(scores, target: 30) {
    print("Первое вхождение 30 на позиции \(first)") // обычно 2
    
    // Можно найти диапазон вручную
    var start = first
    while start > 0 && scores[start - 1] == 30 { start -= 1 }
    
    var end = first
    while end < scores.count - 1 && scores[end + 1] == 30 { end += 1 }
    
    print("Диапазон: \(start)...\(end)") // 2...4
}
```

## Сравнение с бинарным поиском

| Параметр                | Бинарный поиск      | Интерполяционный поиск                       |
| ----------------------- | ------------------- | -------------------------------------------- |
| Средняя сложность       | O(log n)            | O(log log n) — при равномерном распределении |
| Худшая сложность        | O(log n)            | O(n)                                         |
| Требования к данным     | Просто отсортирован | Равномерное распределение                    |
| Надёжность на практике  | ★★★★★               | ★★★☆☆                                        |
| Использование в [[iOS]] | Очень часто         | Редко                                        |

## Итог — когда стоит использовать в 2026 году

| Ситуация                                       | Рекомендация                                 |
| ---------------------------------------------- | -------------------------------------------- |
| Массив равномерно распределённых чисел         | Да, интерполяционный поиск                   |
| Массив ID, timestamp, цены с равномерным шагом | Да                                           |
| Данные неравномерны (экспонента, кластеры)     | Нет → бинарный поиск                         |
| Массив небольшой (< 10 000 элементов)          | Можно [[firstIndex]]`(where:)`               |
| Нужно максимальная надёжность                  | Бинарный поиск                               |
| Хотите удивить на собеседовании                | Интерполяционный + объяснение когда работает |

Интерполяционный поиск — это **красивый**, но **специализированный** алгоритм.  
В большинстве iOS-приложений **бинарный поиск** или `firstIndex(where:)` оказываются практичнее и надёжнее.
